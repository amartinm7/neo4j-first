  (ALICE)<-[:ACTED_IN]-(BOB),
  (ALICE)-[:ACTED_IN]->(BOB),
  (BOB)<-[:ACTED_IN]-(CHARLIE),
  (BOB)-[:ACTED_IN]->(CHARLIE),
  (ALICE)<-[:ACTED_IN]-(CHARLIE),
  (ALICE)-[:ACTED_IN]->(CHARLIE)
RETURN ALICE, BOB , CHARLIE
